# 设置向左或向右扩展
xrandr			# 查看显示屏名称
xrandr --output <屏幕名称> --auto --primary		#设置分辨率
xrandr --output eDP-1-1 --right-of HDMI-0 --auto	#见下一条说明
(xrandr --output <屏幕1> --right-of <屏幕2> --auto	#设置屏幕1 在屏幕2 的右边，如果屏幕1 是主屏幕，则相当于向左扩展，反之则是向右扩展。（主屏幕名称一般包含eDP1字段，扩展屏幕一般包含HDMI字段））
